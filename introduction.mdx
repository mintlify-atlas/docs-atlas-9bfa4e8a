<div className="mt-16 mb-16 max-w-5xl mx-auto px-6">
  <h2 className="text-3xl font-bold text-gray-900 dark:text-white mb-4">
    Quick Start
  </h2>
  <p className="text-base text-gray-600 dark:text-gray-400 mb-8">
    Get your furniture store backend up and running in minutes
  </p>

  <Steps>
    <Step title="Clone the repository">
      Clone the project and navigate to the directory:
      
      ```bash
      git clone https://github.com/furniture-store-utl/backend-furniture-store.git
      cd backend-furniture-store
      ```
    </Step>

    <Step title="Set up virtual environment">
      Create and activate a Python virtual environment:
      
      ```bash
      python -m venv venv
      source venv/bin/activate  # On Windows: venv\Scripts\activate
      ```
    </Step>

    <Step title="Install dependencies">
      Install all required Python packages:
      
      ```bash
      pip install -r requirements.txt
      ```
    </Step>

    <Step title="Configure environment">
      Create a `.env` file based on the template:
      
      ```bash
      cp .env-template .env
      ```
      
      Edit `.env` with your database credentials:
      
      ```bash
      DB_USER=your_user
      DB_PASSWORD=your_password
      DB_HOST=localhost
      DB_PORT=3306
      DB_NAME=furniture_store_db
      ```
    </Step>

    <Step title="Run the application">
      Start the Flask development server:
      
      ```bash
      python run.py
      ```
      
      The server will start at `http://127.0.0.1:5000`
    </Step>
  </Steps>
</div>
